//Comandos (LINUX) para entender el proyecto Mapa deslaves
echo "## node"; node -v
echo "## npm"; npm -v

echo "## project files"; ls -la

echo "## package.json (keys only)"; node -e "const p=require('./package.json'); console.log({name:p.name, version:p.version, type:p.type, scripts:Object.keys(p.scripts||{}), dependencies:Object.keys(p.dependencies||{}), devDependencies:Object.keys(p.devDependencies||{})})"

echo "## deps (top-level)"; npm list --depth=0 --silent

echo "## vite config"; ls -la vite.config.* 2>/dev/null || echo "NO vite.config.*"

echo "## tsconfig files"; ls -la tsconfig*.json 2>/dev/null || echo "NO tsconfig*.json"

echo "## env templates"; ls -la .env* 2>/dev/null || echo "NO .env*"

echo "## tree src (names only)"; find src -maxdepth 6 -type f | sort

echo "## entrypoints present"; \
ls -la index.html 2>/dev/null || echo "NO index.html"; \
ls -la src/main.* 2>/dev/null || echo "NO src/main.*"
